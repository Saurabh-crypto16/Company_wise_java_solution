class Solution {
    public boolean isValid(String s) {
        if(s.length()==1 || s.charAt(0)==')' || s.charAt(0)=='}' || s.charAt(0)==']')   return false;
        Stack<Character> st=new Stack<>();
        for(char c: s.toCharArray()){
            if(c=='(' || c=='{' || c=='['){
                st.push(c);
            }else{
                if(st.isEmpty()){
                    return false;
                }
                else if(c==')'){
                    if(st.peek()=='(')  st.pop();
                    else    return false;
                }else if(c=='}'){
                    if(st.peek()=='{')  st.pop();
                    else    return false;
                }else{
                    if(st.peek()=='[') st.pop();
                    else    return false;
                }
            }
        }
        if(st.isEmpty())    return true;
        return false;
    }
}
