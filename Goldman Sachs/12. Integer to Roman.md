class Solution {
    class Numeral{
        String symbol;
        int val;
        Numeral(String symbol,int val){
            this.symbol=symbol;
            this.val=val;
        }
    }
    
    public String intToRoman(int A) {
        Numeral numerals[]={
            new Numeral("M",1000),
            new Numeral("CM",900),
            new Numeral("D",500),
            new Numeral("CD",400),
            new Numeral("C",100),
            new Numeral("XC",90),
            new Numeral("L",50),
            new Numeral("XL",40),
            new Numeral("X",10),
            new Numeral("IX",9),
            new Numeral("V",5),
            new Numeral("IV",4),
            new Numeral("I",1)
        };
        
        String ans="";
        for(Numeral num: numerals){
            int numberOfSymbol=A/num.val;
            while(numberOfSymbol>0) {
                ans+=num.symbol;
                numberOfSymbol--;
            }
            A%=num.val;
        }
        return ans;
    }
}
