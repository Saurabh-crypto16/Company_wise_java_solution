```javaclass MinStack {
    Stack<Integer> stack = null;
    Stack<Integer> minStack = null;
    public MinStack() {
        stack = new Stack<>();
        minStack = new Stack<>();
    }
    
    public void push(int x) {
        stack.push(x);
        if(minStack.isEmpty() || x <= minStack.peek()) 
            minStack.push(x);
    }
    
    public void pop() {
        if(!stack.isEmpty() && !minStack.isEmpty()) {
            if(minStack.peek().equals(stack.peek()))
                minStack.pop();
            stack.pop();
        }
    }
    
    public int top() {
         if(!stack.isEmpty())
            return stack.peek();
        return -1;
    }
    
    public int getMin() {
        if(!minStack.isEmpty())
            return minStack.peek();
        return -1;
    }
}
 ```
